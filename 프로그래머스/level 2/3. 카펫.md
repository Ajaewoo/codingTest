    class Solution {
      public int[] solution(int brown, int yellow) {
         int[] answer = new int[2];
          int area = brown + yellow;
          int x = 0;
          int y = 0;
          for(y=3;y<area;y++) {
          	if(area%y == 0) {
          		x = area/y;
          		if(x>=y && yellow==(x-2)*(y-2)){
          			answer[0] = x;
          			answer[1] = y;
          		}
          	}
          }
          return answer;
      }
  }
