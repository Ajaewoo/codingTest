```java
    class Solution
    {
        public int solution(int [][]board)
        {
            int answer = 0;
            int x = board.length;
            int y = board[0].length;
            
            if(x<2&&y<2){
                return 1;
            }
            
            for(int i=1;i<x;i++){
                for(int j=1;j<y;j++){
                    if(board[i][j] !=0){                
                        board[i][j] = Math.min(board[i-1][j-1],Math.min(board[i-1][j],board[i][j-1]))+1;
                    }
                    if(answer < board[i][j]) answer = board[i][j];
                }
            }
            return answer*answer;
        }
    }
```
