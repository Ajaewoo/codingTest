    class Solution {
        
    int emtiPlus = 0;
    int sales = 0;
        public int[] solution(int[][] users, int[] emoticons) {
            int[] result = new int[2];
            
            int[] arr = new int[emoticons.length];
            
            comb(arr,0,users,emoticons);
            
            result[0] = emtiPlus;
            result[1] = sales;
            
            return result;
        }
        public void comb(int[] arr, int start, int[][] users, int[] emoticons){
            if(start == arr.length){
                calculation(arr,users,emoticons);
                return;
            }
            for(int i=10; i<=40; i+=10){
                arr[start]=i;
                comb(arr, start+1, users, emoticons);
            }
        }
        
        public void calculation(int[] arr, int[][] users, int[] emoticons){
            int count = 0;
            int sales_t = 0;
            
            for(int[] user:users){
                int discount = user[0];
                int price = user[1];
                int sum = 0;
                
                for(int i=0; i<arr.length; i++){
                    if(arr[i]>=discount){
                        sum+=(emoticons[i]/100)*(100-arr[i]);
                    }
                }
                if(sum>=price){
                    count++;
                }else{
                    sales_t+=sum;
                }
            }
            if(count>emtiPlus){
                emtiPlus = count;
                sales = sales_t;
                return;
            }else if(count == emtiPlus){
                if(sales<sales_t){
                    sales=sales_t;
                }
            }
        }
    }
