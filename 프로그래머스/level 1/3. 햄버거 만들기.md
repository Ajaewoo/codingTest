    import java.util.*;
    class Solution {
        public int solution(int[] ingredient) {
            int result = 0;
            String hamburger = Arrays.toString(ingredient).replaceAll("[^0-9]","");
            while(hamburger.contains("1231")){
                hamburger = hamburger.replaceFirst("1231","");
                result++;
            }
            return result;
        }
    }
